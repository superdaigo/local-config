#+begin_src elisp :results none
  ;; local variables
  (setq-local config-files (list "flake.nix"))
  (setq-local dst-dir (expand-file-name "/etc/nix-darwin"))
  (setq-local tmp-dir (file-name-base buffer-file-name))
  (setq-local backup-prefix (format-time-string "%Y-%m-%d_%H%M%S_"))
  ;; Create directories
  (dolist (dir-name (list dst-dir tmp-dir))
    (make-directory dir-name t))
  ;; Generate file
  (org-babel-tangle-file buffer-file-name)
  ;; Copy current files and deploy
  (dolist (config-file config-files)
    (let ((tmp-file (file-name-concat tmp-dir config-file))
          (bak-file (file-name-concat tmp-dir (concat backup-prefix config-file)))
          (dst-file (file-name-concat dst-dir config-file)))
      ;; Take backup
      (if (file-exists-p dst-file)
          (copy-file dst-file bak-file)
        nil)
      ;; Save new file
      (copy-file tmp-file dst-file t)))
#+end_src

** Rebuild
#+begin_src bash
  sudo darwin-rebuild build --flake /etc/nix-darwin#simple
#+end_src
** Switch to the new build
#+begin_src bash
  sudo darwin-rebuild switch --flake /etc/nix-darwin#simple
#+end_src
** List generations
#+begin_src bash
  sudo darwin-rebuild --list-generations
#+end_src
** Switch to a specific generation
#+begin_src bash
  sudo darwin-rebuild --switch-generation N
#+end_src

Delete generations older than 14days
#+begin_src bash
  sudo nix-collect-garbage --delete-older-than 14d
#+end_src

Delete all generations
#+begin_src bash
  sudo nix-collect-garbage -d
